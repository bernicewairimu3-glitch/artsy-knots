    async function handleSaveProduct(e) {

        return;
      }

      const title = document.getElementById('productTitle').value;
      const description = document.getElementById('productDesc').value;
      const price = parseInt(document.getElementById('productPrice').value);
      const category = document.getElementById('productCategory').value;
      const image = document.getElementById('productImage').value;
      const errorDiv = document.getElementById('productError');

      // Validation
      if (price < 2000 || price > 4000) {
        errorDiv.style.display = 'block';
        errorDiv.textContent = 'Price must be between 2,000 and 4,000 KES';
        return;
      }

      try {
        const productData = {
          title,
          description,
          price,
          category,
          image,
          updatedAt: new Date()
        };

        if (editingProductId) {
          // Update existing product
          await db.collection('websites').doc('artsy-knots').collection('products').doc(editingProductId).update(productData);
        } else {
          // Add new product with likes initialized to 0
          productData.likes = 0;
          productData.createdAt = new Date();
          await db.collection('websites').doc('artsy-knots').collection('products').add(productData);
        }

        closeProductModal();
        loadProductsList();
        loadGallery();
        loadAdminData();
      } catch (error) {
        errorDiv.style.display = 'block';
        errorDiv.textContent = 'Error saving product. Make sure Firebase is configured.';

      }

      try {
        await db.collection('websites').doc('artsy-knots').collection('products').doc(productId).delete();
        loadProductsList();
        loadGallery();
        loadAdminData();
      } catch (error) {
        alert('Error deleting product. Make sure Firebase is configured.');
        console.error('Error:', error);
      }
    }

    // ======================
    // GALLERY (PUBLIC)
    // ======================
    async function loadGallery() {
      const galleryGrid = document.getElementById('galleryGrid');
      galleryGrid.innerHTML = '<div class="loading">Loading gallery...</div>';

      try {
        const querySnapshot = await db.collection('websites').doc('artsy-knots').collection('products').get();

        if (querySnapshot.empty) {
          galleryGrid.innerHTML = '<p style="grid-column: 1/-1; text-align: center; padding: 40px; color: var(--text-muted);">Gallery coming soon...</p>';
          return;
        }

        let html = '';
        querySnapshot.forEach(doc => {
          const product = doc.data();
          const productId = doc.id;
          const likes = product.likes || 0;

          html += `
            <div class="gallery-item" data-category="${product.category}">
              <div class="gallery-item-img">
                <img src="${product.image}" alt="${product.title}" onerror="this.src='https://via.placeholder.com/400x300'">
                <div class="like-badge">${likes} ❤️</div>
              </div>
              <div class="gallery-item-content">
                <div class="gallery-item-title">${product.title}</div>
                <p class="gallery-item-desc">${product.description}</p>
                <div class="gallery-item-price">${product.price} KES</div>
                <span class="gallery-item-category">${product.category}</span>
                <div class="gallery-item-actions">
                  <button class="like-btn" onclick="likeProduct('${productId}', this)" id="like-${productId}">
                    ❤️ Like
                  </button>
                  <button class="whatsapp-btn" onclick="inquireProduct('${product.title}')">
                    WhatsApp me
                  </button>
                </div>
              </div>
            </div>
          `;
        });
        galleryGrid.innerHTML = html;
      } catch (error) {
        galleryGrid.innerHTML = '<div class="error-message" style="grid-column: 1/-1;">Error loading gallery. Make sure Firebase is configured.</div>';
        console.error('Error:', error);
      }
    }

    async function likeProduct(productId, button) {
      try {
        const docRef = db.collection('websites').doc('artsy-knots').collection('products').doc(productId);
        const doc = await docRef.get();

        if (doc.exists) {
          const currentLikes = doc.data().likes || 0;
          await docRef.update({
            likes: currentLikes + 1
          });

          // Update button UI
          button.classList.add('liked');
          button.disabled = true;
          button.textContent = '❤️ Liked!';

          // Update analytics
          const statsRef = db.collection('websites').doc('artsy-knots').collection('analytics').doc('stats');
          const statsDoc = await statsRef.get();
          if (statsDoc.exists) {
            await statsRef.update({
              totalLikes: (statsDoc.data().totalLikes || 0) + 1
            });
          }

          // Reload admin data if admin is viewing
          if (currentUser) {
            loadAdminData();
          }
        }
      } catch (error) {
        console.error('Error liking product:', error);
      }
    }

    function inquireProduct(productName) {
      const message = `Hi Bernice! I'm interested in "${productName}". Can you provide more details?`;
      const whatsappNumber = '254706189808';
      const whatsappURL = `https://wa.me/${whatsappNumber}?text=${encodeURIComponent(message)}`;
      window.open(whatsappURL, '_blank');
    }

    function filterProducts(category, button) {
      const items = document.querySelectorAll('.gallery-item');
      const buttons = document.querySelectorAll('.filter-btn');

      buttons.forEach(btn => btn.classList.remove('active'));
      if (button) {
        button.classList.add('active');
      }

      items.forEach(item => {
        const itemCategory = item.getAttribute('data-category');
        if (category === 'all' || itemCategory === category) {
          item.classList.remove('hidden');
        } else {
          item.classList.add('hidden');
        }
      });
    }
